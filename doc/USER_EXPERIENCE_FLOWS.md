### 4.4 Scanner Error Handling

```typescript
// components/BarcodeScanner.tsx

type ScannerState = 
  | { status: 'ready' }
  | { status: 'detected'; barcode: string }
  | { status: 'loading'; barcode: string }
  | { status: 'found'; food: FoodItem }
  | { status: 'not_found'; barcode: string }
  | { status: 'error'; message: string }
  | { status: 'offline'; barcode: string };

function handleBarcodeScan(barcode: string) {
  setState({ status: 'detected', barcode });
  
  // Haptic feedback
  Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium);
  
  // Lookup
  setTimeout(async () => {
    setState({ status: 'loading', barcode });
    
    try {
      const food = await getProductByBarcode(barcode);
      
      if (food) {
        setState({ status: 'found', food });
      } else if (!(await isOnline())) {
        setState({ status: 'offline', barcode });
      } else {
        setState({ status: 'not_found', barcode });
      }
    } catch (error) {
      setState({ status: 'error', message: 'Something went wrong. Try again.' });
    }
  }, 300); // Brief delay for visual feedback
}
```
